Here's the design for the RESTful API for the Product Wishlist microservice:

1. **OpenAPI 3.0 specification in YAML**:

```yaml
openapi: 3.0.0
info:
  title: Product Wishlist API
  version: 1.0.0

servers:
  - url: http://api.example.com/v1

paths:
  /wishlists:
    get:
      summary: Get the authenticated user's wishlist
      responses:
        '200':
          description: Successful response
          content:
            application/json:    
              schema:
                $ref: '#/components/schemas/Wishlist'
    post:
      summary: Create a new wishlist
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/NewWishlist'
      responses:
        '201':
          description: Successful response
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Wishlist'
  /wishlists/{wishlistId}/products:
    get:
      summary: Get products in a wishlist
      parameters:
        - in: path
          name: wishlistId
          required: true
          schema:
            type: string
            format: uuid
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              schema:
                type: array
                items:
                  $ref: '#/components/schemas/ProductItem'
    post:
      summary: Add a product to a wishlist
      parameters:
        - in: path
          name: wishlistId
          required: true
          schema:
            type: string
            format: uuid
      requestBody:
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/NewProductItem'
      responses:
        '201':
          description: Successful response
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/ProductItem'
    delete:
      summary: Remove a product from a wishlist
      parameters:
        - in: path
          name: wishlistId
          required: true
          schema:
            type: string
            format: uuid
        - in: query
          name: productId
          required: true
          schema:
            type: string
            format: uuid
      responses:
        '204':
          description: Successful response

components:
  schemas:
    Wishlist:
      type: object
      properties:
        id:
          type: string
          format: uuid
        name:
          type: string
        products:
          type: array
          items:
            $ref: '#/components/schemas/ProductItem'
    NewWishlist:
      type: object
      properties:
        name:
          type: string
    ProductItem:
      type: object
      properties:
        id:
          type: string
          format: uuid
        name:
          type: string
        description:
          type: string
    NewProductItem:
      type: object
      properties:
        productId:
          type: string
          format: uuid
```

3. **docker-compose.yml for the service and PostgreSQL database**:

```yaml
version: '3'

services:
  db:
    image: postgres:13
    environment:
      POSTGRES_DB: wishlist
      POSTGRES_USER: wishlist_user
      POSTGRES_PASSWORD: wishlist_password
    volumes:
      - db-data:/var/lib/postgresql/data
    ports:
      - "5432:5432"

  api:
    build: .
    environment:
      DB_HOST: db
      DB_USER: wishlist_user
      DB_PASSWORD: wishlist_password
      DB_NAME: wishlist
    ports:
      - "8080:8080"
    depends_on:
      - db

volumes:
  db-data:
```

This docker-compose.yml file sets up a PostgreSQL database and the Product Wishlist API service. The database is configured with a `wishlist` database, `wishlist_user` as the user, and `wishlist_password` as the password. The API service is built from the Dockerfile in the current directory and is configured to connect to the PostgreSQL database using the provided environment variables.
